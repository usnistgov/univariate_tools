 70def least_squares_fit(function,xdata,ydata,initial_parameters):
 71    """Returns a parameter list after fitting the data x_data,y_data
 72    with the function in the form of f(parameter_list,x)"""
 73    error_function=lambda a, xdata, ydata:function(a,xdata)-ydata
 74    a,success=scipy.optimize.leastsq(error_function, initial_parameters,args=(np.array(xdata),np.array(ydata)))
 75    return a

 77def calculate_residuals(fit_function,a,xdata,ydata):
 78    """Given the fit function, a parameter vector, xdata, and ydata returns the residuals as [x_data,y_data]"""
 79    output_x=xdata
 80    output_y=[fit_function(a,x)-ydata[index] for index,x in enumerate(xdata)]
 81    return [output_x,output_y]

 83def build_modeled_data_set(x_list,function_list):
 84    """build_modeled_data_set takes an input independent variable and a list
 85    of  functions and returns a data set of the form [..[xi,f1(xi),..fn(xi)]]
 86    it is meant to create a modeled data set. Requires that the list of functions is callable f(x) is defined"""
 87    out_data=[]
 88    for x in x_list:
 89        new_row=[]
 90        new_row.append(x)
 91        for function in function_list:
 92            if type(function(x)) in [np.ndarray,'numpy.array']:
 93                # to list is needed if the function returns np.array
 94                new_row.append(function(x).tolist())
 95            else:
 96                new_row.append(function(x))
 97        out_data.append(new_row)
 98    return out_data

102class FunctionalModel(object):
103    """FittingModel is a class that holds a fitting function, it uses sympy to provide
104     symbolic manipulation of the function and formatted output. If called it acts like a
105     traditional python function. Initialize the class with parameters, variables and an equation.
106     Ex. line=FunctionalModel(variables='x', parameters='m b', equation='m*x+b'), to call as a function
107     the parameters must be set. line(m=2,b=1,x=1)
108     or
109     line.set_parameters(m=1,b=2)
110     line(1). To fit data use the .fit_data(x_data,y_data) method. This will automatically set the parameters to
111     the fit values."""
112
113    def __init__(self, **options):
114        """Initializes the FunctionalModel, note if the equation passed has a special function bessel,etc.
115        it will find it"""
116        defaults = {"parameters": None,
117                    "variables": None,
118                    "equation": None,
119                    "parameter_values": {},
120                    "special_function": False}
121        self.options = {}
122        for key, value in defaults.items():
123            self.options[key] = value
124        for key, value in options.items():
125            self.options[key] = value
126        # fix any lists
127        for item in ["parameters", "variables"]:
128            if isinstance(self.options[item], str):
129                self.options[item] = re.split(r"\s+", self.options[item])
130            self.__dict__[item] = self.options[item]
131            self.options[item] = None
132        self.special_function = self.options["special_function"]
133        self.equation = sympy.sympify(self.options["equation"])
134        special_function_keys = list(LAMBDIFY_MODULES[1].keys())
135        for key in special_function_keys:
136            if re.search(key, str(self.options["equation"])):
137                self.special_function = True
138        if self.special_function:
139            self.function = sympy.lambdify(self.parameters + self.variables, self.equation, modules=LAMBDIFY_MODULES)
140        else:
141            self.function = sympy.lambdify(self.parameters + self.variables, self.equation, "numpy")
142        self.parameter_values = self.options["parameter_values"]
143        self.options["parameter_values"] = {}
144
145    def __call__(self, *args, **keywordargs):
146        """Controls the behavior when called as a function"""
147        return self.function(*args, **keywordargs)

149    def set_parameters(self, parameter_dictionary=None, **parameter_dictionary_keyword):
150        """Sets the parameters to values in dictionary"""
151        if parameter_dictionary is None:
152            try:
153                parameter_dictionary = parameter_dictionary_keyword
154            except:
155                pass
156        self.parameter_values = parameter_dictionary
157        if self.special_function:
158            self.function = np.vectorize(lambda x: sympy.N(sympy.lambdify(self.variables,
159                                                                          self.equation.subs(self.parameter_values),
160                                                                          modules=LAMBDIFY_MODULES)(x)))
161        else:
162            self.function = sympy.lambdify(self.variables, self.equation.subs(self.parameter_values),
163                                           modules=LAMBDIFY_MODULES)
164
165    def clear_parameters(self):
166        """Clears the parameters specified by set_parameters"""
167        self.function = sympy.lambdify(self.parameters + self.variables, self.equation, modules=LAMBDIFY_MODULES)
168        self.parameter_values = {}

170    def fit_data(self, x_data, y_data, **options):
171        """Uses the equation to fit the data, after fitting the data sets the parameters.
172        """
173        defaults = {"initial_guess": {parameter: 0.0 for parameter in self.parameters}, "fixed_parameters": None}
174        self.fit_options = {}
175        for key, value in defaults.items():
176            self.fit_options[key] = value
177        for key, value in options.items():
178            self.fit_options[key] = value
179
180        def fit_f(a, x):
181            self.clear_parameters()
182            input_list = []
183            for parameter in a:
184                input_list.append(parameter)
185            if self.special_function:
186                if isinstance(x, list):
187                    out = np.array(list(map(lambda y: sympy.N(self.function(*(input_list + [y]))), x)),
188                                   dtype=np.float64)
189                    return out
190                elif isinstance(x, np.ndarray):
191                    out = np.array(list(map(lambda y: sympy.N(self.function(*(input_list + [y]))), x.tolist())),
192                                   dtype=np.float64)
193                    return out
194                else:
195                    out = sympy.N(self.function(*input_list))
196                    return out
197            else:
198                input_list.append(x)
199                return self.function(*input_list)
200
201        # this needs to be reflected in fit_parameters
202        a0 = []
203        for key in self.parameters[:]:
204            a0.append(self.fit_options["initial_guess"][key])
205        a0 = np.array(a0)
206        result = least_squares_fit(fit_f, x_data, y_data, a0)
207        fit_parameters = result.tolist()
208        fit_parameter_dictionary = {parameter: fit_parameters[index] for index, parameter in enumerate(self.parameters)}
209        self.set_parameters(fit_parameter_dictionary)

314    def d(self, respect_to=None, order=1):
315        """Takes the derivative with respect to variable or parameter provided or defaults to first variable"""
316        if respect_to is None:
317            respect_to = self.variables[0]
318        equation = self.equation.copy()
319        for i in range(order):
320            equation = sympy.diff(equation, respect_to)
321        return FunctionalModel(parameters=self.parameters[:], variables=self.variables[:], equation=str(equation))

323    def integrate(self, respect_to=None, order=1,limits = None):
324        """Integrates with respect to variable or parameter provided or defaults to first variable.
325        Does not add a constant of integration."""
326        if respect_to is None:
327            respect_to = self.variables[0]
328        equation = self.equation.copy()
329        for i in range(order):
330            if limits:
331                value = sympy.integrate(equation,(respect_to,limits[0],limits[1]))
332                return value
333            else:
334                equation = sympy.integrate(equation, sympy.symbols(respect_to))
335        return FunctionalModel(parameters=self.parameters[:], variables=self.variables[:], equation=str(equation))

337    def limit(self, variable_or_parameter, point):
338        """Finds the symbolic limit of the FunctionalModel for the variable or parameter approaching point"""

340        parameters = self.parameters[:]
341        variables = self.variables[:]
342        return FunctionalModel(equation=equation, variables=variables, parameters=parameters)

346    """A class that simulates data. It creates a data set from a FunctionalModel with the parameters set,
347    and an optional output noise. The attribute self.x has the x data and self.data has the result. The simulator may be

355                   "model":None,
356                   "variable_min":None,

358                   "number_points":None,
359                   "variable_step":None,
360                   "output_noise_type":None,
361                   "output_noise_width":None,
362                   "output_noise_center":None,
363                   "output_noise_amplitude":1.,
364                   "random_seed":None,
365                   "x":np.array([])}
366        self.options={}
367        for key,value in defaults.items():
368            self.options[key]=value
369        for key,value in options.items():
370            self.options[key]=value
371        # set the self.model attribute
372        if self.options["model"]:
373            self.model=self.options["model"]
374        else:
375            # try and create the model from the options
376            try:
377                self.model=FunctionalModel(variables=self.options["variables"],
378                                           parameters=self.options["parameters"],
379                                          equation=self.options["equation"])
380            except:
381                print("Could not form a model from the information given, either model has to be specified or"
382                     "parameters, variables and equation has to be specified")
383                # todo: make an error specific to this case
384                raise
385        if self.options["parameter_values"]:
386            self.model.set_parameters(self.options["parameter_values"])
387        self.x=self.options["x"]
388        self.random_seed=self.options["random_seed"]
389        output_noise_names=["type","center","width","amplitude"]
390        for index,output_noise_name in enumerate(output_noise_names):
391            self.__dict__["output_noise_{0}".format(output_noise_name)]=self.options["output_noise_{0}".format(output_noise_name)]
392        self.set_output_noise()
393        if self.options["variable_min"] and self.options["variable_max"]:
394            self.set_x(variable_min=self.options["variable_min"],
395                       variable_max=self.options["variable_max"],
396                       number_points=self.options["number_points"],
397                       variable_step=self.options["variable_step"])

402        self.clear_parameters=self.model.clear_parameters

407    def set_x(self, variable_min=None, variable_max=None,number_points=None,variable_step=None):
408        """Sets the dependent variable values, min, max and number of points or step"""
409        if [variable_min,variable_max]==[None,None]:
410            self.x=np.array([])
411        elif isinstance(variable_min,list):
412            self.x=variable_min
413        else:
414            if variable_step:
415                number_points=(variable_max-variable_min)/variable_step
416            self.x=np.linspace(variable_max,variable_min,number_points)
417        self.set_output_noise()

420    def set_output_noise(self,output_noise_type=None,output_noise_center=None,output_noise_width=None,output_noise_amplitude=1.):
421        """Set the output noise distrubution. Possible types are gaussian, uniform, triangular, lognormal, with the
422        assumption all are symmetric
423        """
424        output_noise_characteristics=[output_noise_type,output_noise_center,output_noise_width,output_noise_amplitude]
425        output_noise_names=["type","center","width","amplitude"]
426        for index,output_noise_characteristic in enumerate(output_noise_characteristics):
427            if output_noise_characteristic:
428                self.__dict__["output_noise_{0}".format(output_noise_names[index])]=output_noise_characteristic
429        if self.output_noise_type is None or not self.x.any():
430            self.output_noise=np.array([])
431        else:
432            # set the random seed
433            np.random.seed(self.random_seed)
434
435            # now handle the output types, all in np.random
436            if re.search("gauss|normal",self.output_noise_type,re.IGNORECASE):
437                self.output_noise=output_noise_amplitude*np.random.normal(self.output_noise_center,
438                                                                    self.output_noise_width,len(self.x))
439            elif re.search("uni|square|rect",self.output_noise_type,re.IGNORECASE):
440                self.output_noise=output_noise_amplitude*np.random.uniform(self.output_noise_center-self.output_noise_width/2,
441                                                                     self.output_noise_width+self.output_noise_width/2,
442                                                                     len(self.x))
443            elif re.search("tri",self.output_noise_type,re.IGNORECASE):
444                self.output_noise=output_noise_amplitude*np.random.triangular(self.output_noise_center-self.output_noise_width/2,
445                                                                     self.output_noise_center,
446                                                                     self.output_noise_width+self.output_noise_width/2,
447                                                                    len(self.x))
448        self.set_data()

484class Multicosine(FunctionalModel):
485    """Multicosine creates a Functional Model of the form f(t) = A_1*cos(2*pi*frequency_list[0]*t+phi_1)+
486    ... A_N*cos(2*pi*frequency_list[N]*t+phi_N). It requires a frequency_list to be passed on creation"""
487    def __init__(self,frequency_list):
488        # Constructing the equation string to pass to FunctionalModel
489        number_terms = len(frequency_list)
490        fit_function = ""
491        for i in range(number_terms):
492            if i < number_terms - 1:
493                fit_function = fit_function + "A_{0}*cos(2*pi*{1}*t+phi_{0})+".format(i + 1, frequency_list[i])
494            else:
495                fit_function = fit_function + "A_{0}*cos(2*pi*{1}*t+phi_{0})".format(i + 1, frequency_list[i])
496        # Construct the parameter List
497        parameter_list = ["A_{0}".format(i + 1) for i in range(number_terms)] + ["phi_{0}".format(i + 1) for i in
498                                                                                 range(number_terms)]
499        FunctionalModel.__init__(self,parameters=parameter_list,variables="t",equation=fit_function)
